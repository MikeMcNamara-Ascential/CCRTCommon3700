#------------------------------------------------------------------------------------------
# Define the basename of our executable (for us only...not required by CMake)
#------------------------------------------------------------------------------------------
SET(EXE_NAME WorkCellController)

#------------------------------------------------------------------------------------------
# Need to add core include directory
#------------------------------------------------------------------------------------------
INCLUDE_DIRECTORIES(${CORE_PLANT_HOST_IN_INC_DIR})

#------------------------------------------------------------------------------------------
# Link against the core sysmon lib
#------------------------------------------------------------------------------------------
LINK_DIRECTORIES(${CORE_PLANT_HOST_IN_LIB_DIR})
LINK_LIBRARIES(PlantHostIn)

#------------------------------------------------------------------------------------------
# Tell CMake what subdirectories to recurse into
#------------------------------------------------------------------------------------------
ADD_SUBDIRECTORY(Configuration)

#------------------------------------------------------------------------------------------
# Define the source files needed to build the executable
#------------------------------------------------------------------------------------------
FILE(GLOB WCC_SOURCES *.cpp)

#------------------------------------------------------------------------------------------
# Tell CMake what source file to build
#------------------------------------------------------------------------------------------
ADD_EXECUTABLE(${EXE_NAME}.${CMAKE_EXECUTABLE_SUFFIX} ${WCC_SOURCES})

#------------------------------------------------------------------------------------------
# Make sure that the services layer lib is built first
#------------------------------------------------------------------------------------------
INCLUDE(${CCRT_ROOT}/LibServices.cmake)

#------------------------------------------------------------------------------------------
# Make sure the executable has the correct extension
#------------------------------------------------------------------------------------------
INCLUDE(${CCRT_ROOT}/ExeQxe.cmake)

#------------------------------------------------------------------------------------------
# Tell CMake where to install the executable
#------------------------------------------------------------------------------------------
INSTALL_TARGETS(${CUSTOMER_SERVICES_BIN_DEPLOY_DIR} ${EXE_NAME}.qxe)
