#------------------------------------------------------------------------------------------
# Create a list of all the cpp files in this directory needed to create the executable
#------------------------------------------------------------------------------------------
FILE(GLOB GRYPH_J1850_SOURCES *.cpp)

#------------------------------------------------------------------------------------------
# Define the basename of our executable (for us only...not required by CMake)
#------------------------------------------------------------------------------------------
SET(EXE_NAME GryphonJ1850CommProxy)

#------------------------------------------------------------------------------------------
# Add an include path for the DG supplied headers
#------------------------------------------------------------------------------------------
INCLUDE_DIRECTORIES(
	${CORE_GRYPHON_INC_DIR} 
	${CORE_GRYPHON_INC_DIR}/hw 
	${CORE_COMMPROXY_INC_PATH}
	${CORE_RESMANAGER_INC_PATH}
	)

#------------------------------------------------------------------------------------------
# Link against the core Gryph lib
#------------------------------------------------------------------------------------------
LINK_LIBRARIES(GryphCommon)

#------------------------------------------------------------------------------------------
# Tell CMake what source file to build
#------------------------------------------------------------------------------------------
CREATE_QXE(${EXE_NAME} ${GRYPH_J1850_SOURCES})

#------------------------------------------------------------------------------------------
# Add a QNX use message to the executable
#------------------------------------------------------------------------------------------
ADD_USE_MSSG(${EXE_NAME} ${CMAKE_CURRENT_SOURCE_DIR}/GryphonJ1850ProxyMain.cpp)

#------------------------------------------------------------------------------------------
# Tell CMake where to install the executable
#------------------------------------------------------------------------------------------
INSTALL_TARGETS(${SERVICES_LAYER_INSTALL_DIR} ${EXE_NAME})
